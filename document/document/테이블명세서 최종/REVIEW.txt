CREATE TABLE REVIEW(
NUM NUMBER PRIMARY KEY,
WRITER VARCHAR2(30),
SUBJECT VARCHAR2(100),
CONTENT_P VARCHAR2(4000),
CONTENT_M VARCHAR2(4000),
IP VARCHAR2(30),
ACCOMODATION VARCHAR2(50),
LODGE_DATE VARCHAR2(50),
WRITE_DATE VARCHAR2(50),
IMAGE VARCHAR2(4000),
RECOM NUMBER,
REPLY NUMBER
);

CREATE SEQUENCE REVIEW_SEQ;
