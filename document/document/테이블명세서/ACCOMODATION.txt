create sequence accomodation_seq;

create table accomodation(
num number primary key,
category_accomodation varchar2(50) not null,
category_room varchar2(50) not null,
homepage varchar2(100) not null,
accomodation_name varchar2(100) not null,
country varchar2(40) not null,
city varchar2(50) not null,
address varchar2(100) not null,
facility varchar2(500) not null,
image varchar2(500) not null,
price number not null,
content varchar2(4000),
policy varchar2(500),
checkIn_date varchar2(50),
checkOut_date varchar2(50),
rooms number default 1,
adults number default 1,
children number default 0,
review_num number not null,
foreign key(review_num) references review(num),
grade_num number not null,
foreign key(grade_num) references grade(num));
