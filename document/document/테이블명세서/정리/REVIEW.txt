create table review
(num number primary key,
writer varchar2(30),
subject varchar2(100),
content_p varchar2(4000),
content_m varchar2(4000),
ip varchar2(30),
accommodation varchar2(50),
lodge_date varchar2(50),
write_date varchar2(50),
image varchar2(4000),
recom number,
reply number);

create sequence review_seq;

create table grade
(num number primary key,
accommodation varchar2(50),
review number,
clean number,
comfortable number,
location number,
facilities number,
kind number,
value number,
wifi number);

create sequence grade_seq;
